Everything is installed now. Let's set up some scripts for you to run your project.

<NPMScript name="develop" command="gatsby develop" />
<NPMScript name="dev:hard" command="gatsby clean && gatsby develop" />
<NPMScript name="build" command="gatsby build" />
<NPMScript name="build:hard" command="gatsby clean && gatsby build" />

---

Next we install the plugins in gatsby-config.js and add a little bit of configuration.

<GatsbyPlugin
    name="gatsby-source-filesystem"
    key="homepage"
    options={{
        name: `homepage`,
        path: `${__dirname}/src/homepage/`
    }} />
<GatsbyPlugin
    name="gatsby-source-filesystem"
    key="articles"
    options={{
        name: `articles`,
        path: `${__dirname}/src/articles/`
    }} />
<GatsbyPlugin name="gatsby-plugin-theme-ui" />
<GatsbyPlugin name="gatsby-plugin-sharp" />
<GatsbyPlugin name="gatsby-transformer-sharp" />
<GatsbyPlugin name="@jbolda/gatsby-theme-homepage"
    options={{ showArticlesOnHomepage: true }} />
<GatsbyPlugin name="@jbolda/gatsby-theme-articles" options={{
    contents: [
          { contentPath: "articles" },
        ]
    }}/>
<GatsbyPlugin name="gatsby-plugin-mdx" options={{
    gatsbyRemarkPlugins: [
        {
            resolve: `gatsby-remark-images`,
            options: {
                maxWidth: 590
            }
        }
    ]
    }} />
<GatsbyPlugin name="gatsby-plugin-react-helmet" />
<GatsbyPlugin name="gatsby-plugin-netlify" />

---

Config is all set. Let's shadow the theme file to allow you to customize the look and feel of your site.

<GatsbyShadowFile theme="gatsby-plugin-theme-ui" path="src/index.js" />

---
