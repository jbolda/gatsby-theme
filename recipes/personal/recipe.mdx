---

Next we install the plugins in gatsby-config.js and add a little bit of configuration.

First the filesystem source for the homepage.

<GatsbyPlugin
    name="gatsby-source-filesystem"
    key="homepage"
    options={{
        name: "homepage",
        path: "src/homepage/"
    }} />
<GatsbyPlugin
    name="gatsby-source-filesystem"
    key="articles"
    options={{
        name: "articles",
        path: "src/articles/"
    }} />
<GatsbyPlugin name="gatsby-plugin-theme-ui" />
<GatsbyPlugin name="gatsby-plugin-sharp" />
<GatsbyPlugin name="gatsby-transformer-sharp" />
<GatsbyPlugin name="@jbolda/gatsby-theme-homepage"
    options={{ showArticlesOnHomepage: true }} />
<GatsbyPlugin name="@jbolda/gatsby-theme-articles" options={{
    contents: [
          { contentPath: "articles" },
        ]
    }}/>
<GatsbyPlugin name="gatsby-plugin-mdx" options={{
    gatsbyRemarkPlugins: [
        {
            resolve: `gatsby-remark-images`,
            options: {
                maxWidth: 590
            }
        }
    ]
    }} />
<GatsbyPlugin name="gatsby-plugin-react-helmet" />
<GatsbyPlugin name="gatsby-plugin-netlify" />
